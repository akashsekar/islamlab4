import pandas as pd
import matplotlib.pyplot as plt

data = pd.read_csv('penguins.csv')

data['BMI'] = (data['body_mass_g'] / ((data['flipper_length_mm'] / 1000) ** 2))
mean_bmi_by_gender = data.groupby('sex')['BMI'].mean()
plt.figure(figsize=(8, 6))
mean_bmi_by_gender.plot(kind='bar', color=['yellow', 'green'], alpha=0.7)
plt.title('mean BMI bygender penguins')
plt.xlabel('gender')
plt.ylabel('mean BMI')
plt.xticks(rotation=0) 
plt.show()
